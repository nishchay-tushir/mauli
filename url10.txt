url = input("Enter the URL - ")
# Split the URL by "://" to separate protocol and the rest of the URL
parts = url.split("://")
if len(parts) == 2:
protocol, rest = parts
else:
print("Invalid URL format")
exit()
# Split the rest of the URL by "/" to get domain, port, and path
split_rest = rest.split("/")
domain = split_rest[0]
# Check if there is a port specified
if ":" in domain:
domain, port = domain.split(":")
else:
port = "Not specified"
# Get the path
path = "/".join(split_rest[1:])
print(f"Protocol - {protocol}")
print(f"Domain - {domain}")
print(f"Port - {port}")
print(f"Path - {path}")
